ruleModel (HelloWorld)


Model:
  greetings+=Greeting*

  ( keywordsExample = KeywordsExample )?
;

ruleKeywordsExample (HelloWorld)


/**
 * This example illustrated keyword annotation.
 * Predicates are propagated into alternative before Xtext-generated actions.
 */
KeywordsExample:

;
